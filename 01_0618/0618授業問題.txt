//伊藤 遠藤 佐藤 鈴木 松尾 //ul要素はあらかじめhtml内に書いておく

var array = [];
array = ["伊藤","遠藤","佐藤","鈴木","松尾"]; 
var ulList = document.querySelector('ul');//ul要素をhtmlから引っ張ってこい

//4ではなく、lengthで数字を返す
for ( let i = 0; i < array.length; i++){ //lengthは配列内にある数を返す言葉 //４以下になったら処理を終えろ
  var list = document.createElement('li');
  var addList = ulList.appendChild(list);
  addList.innerText = array[i];   //liタグの中に五人の苗字を表示させる //addListを指定しないと、ul直下に反映されないから

if ( i % 2 == 0){
  addList.style = "color: red"; //配列は0からカウントされることを忘れない!!!!!
  } 
else{
  addList.style = "color: blue";
  }

}


//このパターンでも出来る
if ( i % 2 == 0){
  addList.className = "red"; //文字が赤くなる
  } 
else{
  addList.className = "blue";
  }
